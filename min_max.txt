def minimax(depth, node_index, is_max, values, max_depth, alpha, beta):
    # Base case: leaf node reached
    if depth == max_depth:
        return values[node_index]

    if is_max:
        best = float('-inf')
        # Check both left (index*2) and right (index*2 + 1) children
        for i in range(2):
            val = minimax(depth + 1, node_index * 2 + i, False, values, max_depth, alpha, beta)
            best = max(best, val)
            alpha = max(alpha, best) # alpha --> best max value
            # Pruning
            if beta <= alpha:
                break
        return best
    
    else:
        best = float('inf') # Corrected: Minimizer starts at +infinity
        # Check both left (index*2) and right (index*2 + 1) children
        for i in range(2):
            val = minimax(depth + 1, node_index * 2 + i, True, values, max_depth, alpha, beta)
            best = min(best, val)
            beta = min(beta, best) # beta --> best min value
            # Pruning
            if beta <= alpha:
                break
        return best

values = [4, 5, 3, 7, -1, 0, 8, 9]
# Call with alpha as -inf and beta as +inf
result = minimax(0, 0, True, values, 3, float('-inf'), float('inf'))

print('Optimal Value:', result)
